## Design Context (locked decisions)

Inherited from WC Labor Model Slice 0.

**The auto-compute method is `_recomputeZoneRepairLaborIfAuto` at
`storage.ts:7673`.** It is called from `createWetCheckFinding`
(`:7924`), `updateWetCheckFinding` (`:7973`), and the various reset

note 5). Fix it here.

per company on startup." Slice 1 Step A locates the actual seed
application call — find that code path and verify whether it runs
on company-create or only on app boot.

**(2) Write the seeder.** New script:

```typescript
// artifacts/api-server/src/scripts/seed-issue-type-configs.ts
//
// Idempotent. For each company, ensures every row in
// ISSUE_TYPE_CONFIG_SEEDS exists. Inserts missing rows; never updates
// existing rows (a company may have legitimately customized
// defaultLaborHours).

import { db } from "../db";
import { issueTypeConfigs, companies } from "@workspace/db/schema";
import { eq, and, inArray } from "drizzle-orm";
import { ISSUE_TYPE_CONFIG_SEEDS } from "../seeds/issue-type-configs";

export interface SeedResult {
  companiesProcessed: number;
  rowsInserted: number;
  rowsSkipped: number;
}

export async function seedIssueTypeConfigsForAllCompanies(): Promise<SeedResult> {
  const allCompanies = await db.select({ id: companies.id }).from(companies);
  let rowsInserted = 0;
  let rowsSkipped = 0;
  for (const c of allCompanies) {
    const result = await seedIssueTypeConfigsForCompany(c.id);
    rowsInserted += result.rowsInserted;
    rowsSkipped += result.rowsSkipped;
  }
  return { companiesProcessed: allCompanies.length, rowsInserted, rowsSkipped };
}

export async function seedIssueTypeConfigsForCompany(companyId: number): Promise<{ rowsInserted: number; rowsSkipped: number }> {
  const existing = await db
    .select({ issueType: issueTypeConfigs.issueType })
    .from(issueTypeConfigs)
    .where(eq(issueTypeConfigs.companyId, companyId));
  const existingKeys = new Set(existing.map((r) => r.issueType));
  const toInsert = ISSUE_TYPE_CONFIG_SEEDS.filter((s) => !existingKeys.has(s.issueType));
  if (toInsert.length === 0) return { rowsInserted: 0, rowsSkipped: ISSUE_TYPE_CONFIG_SEEDS.length };
  await db.insert(issueTypeConfigs).values(
    toInsert.map((s) => ({
      companyId,
      issueType: s.issueType,
      issueGroup: s.issueGroup,
      displayLabel: s.displayLabel,
      defaultLaborHours: s.defaultLaborHours,
      partCategoryFilter: s.partCategoryFilter,
      sortOrder: s.sortOrder,
    })),
  );
  return { rowsInserted: toInsert.length, rowsSkipped: existingKeys.size };
}

// CLI usage: `npx tsx scripts/seed-issue-type-configs.ts` from the api-server.
if (require.main === module) {
  seedIssueTypeConfigsForAllCompanies().then((r) => {
    console.log(`Seeded: ${JSON.stringify(r)}`);
    process.exit(0);
  }).catch((e) => { console.error(e); process.exit(1); });
}
```

#### B2b — Normalize `issueType` strings

If Slice 1's audit shows mismatched keys (e.g., findings carry

Add a helper at the top of `storage.ts` (or in a shared util):

```typescript
function normalizeIssueTypeKey(s: string): string {
  return s.trim().toLowerCase().replace(/[\s-]+/g, "_");
}
```

```typescript
const ISSUE_TYPE_ALIASES: Record<string, string> = {
  "nozzle_replace": "nozzle_replacement",
  "head_replace": "head_replacement",
  // Add from audit findings
};

function resolveIssueTypeKey(s: string): string {
  const normalized = normalizeIssueTypeKey(s);
  return ISSUE_TYPE_ALIASES[normalized] ?? normalized;
}
```
