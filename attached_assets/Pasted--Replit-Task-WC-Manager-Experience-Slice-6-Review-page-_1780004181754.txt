#### (1) Build the DismissibleHelp primitive

```typescript
// artifacts/irrigopro/src/components/shared/dismissible-help.tsx
//
// Slice 6 — Dismissible per-user help tooltip / popover. Persistence:
// per-user localStorage key. The toggle in the page header re-summons
// dismissed guides (handled by the parent page, not this primitive).
//
// Used by:
//   - Wet Check Review page (Slice 6)
//   - Unified Wet Checks list page (Slice 7)

import { useEffect, useState } from "react";
import { X, HelpCircle } from "lucide-react";

export interface DismissibleHelpProps {
  /** Unique id used as the localStorage key suffix. */
  guideId: string;
  /** When true, the guide is rendered. When false, it is fully unmounted. */
  open?: boolean;
  /** Position relative to the trigger. Default: 'bottom'. */
  side?: "top" | "bottom" | "left" | "right";
  /** Variant — affects color treatment. Default: 'info'. */
  variant?: "info" | "success" | "warning";
  /** Body content. Plain text or React node. */
  children: React.ReactNode;
  /** Called when the user dismisses the guide. */
  onDismiss?: () => void;
  /** When true (default), localStorage is consulted on mount to decide visibility. */
  persistDismissal?: boolean;
}

const KEY_PREFIX = "irrigopro:help-dismissed:";

export function isHelpDismissed(guideId: string): boolean {
  try {
    return localStorage.getItem(KEY_PREFIX + guideId) === "true";
  } catch {
    return false;
  }
}

export function resetHelpDismissal(guideId: string): void {
  try {
    localStorage.removeItem(KEY_PREFIX + guideId);
  } catch {}
}

export function DismissibleHelp({
  guideId,
  open = true,
  side = "bottom",
  variant = "info",
  children,
  onDismiss,
  persistDismissal = true,
}: DismissibleHelpProps) {
  const [dismissed, setDismissed] = useState(() =>
    persistDismissal ? isHelpDismissed(guideId) : false
  );
  // Re-check localStorage when guideId changes (page navigation).
  useEffect(() => {
    if (persistDismissal) setDismissed(isHelpDismissed(guideId));
  }, [guideId, persistDismissal]);

  if (dismissed || !open) return null;

  const dismiss = () => {
    if (persistDismissal) {
      try { localStorage.setItem(KEY_PREFIX + guideId, "true"); } catch {}
    }
    setDismissed(true);
    onDismiss?.();
  };

  // Close on Esc.
  useEffect(() => {
    const handler = (e: KeyboardEvent) => {
      if (e.key === "Escape") dismiss();
    };
    window.addEventListener("keydown", handler);
    return () => window.removeEventListener("keydown", handler);
  }, []);

  const variantClasses = {
    info: "bg-blue-50 border-blue-200 text-blue-900",
    success: "bg-green-50 border-green-200 text-green-900",
    warning: "bg-amber-50 border-amber-200 text-amber-900",
  }[variant];

  return (
    <div
      role="status"
      data-testid={`help-${guideId}`}
      data-side={side}
      className={`rounded-md border px-3 py-2 flex items-start gap-2 text-sm ${variantClasses}`}
    >
      <HelpCircle className="w-4 h-4 mt-0.5 flex-shrink-0" aria-hidden />
      <div className="flex-1">{children}</div>
      <button
        type="button"
        onClick={dismiss}
        aria-label="Dismiss help"
        className="opacity-70 hover:opacity-100"
        data-testid={`help-dismiss-${guideId}`}
      >
        <X className="w-4 h-4" aria-hidden />
      </button>
    </div>
  );
}
```

Tests for the primitive: assert that (a) the guide renders when not
dismissed, (b) clicking dismiss writes the localStorage key and
unmounts, (c) `Esc` dismisses, (d) `isHelpDismissed` reflects state,
(e) `resetHelpDismissal` clears the key.

#### (4) Keyboard shortcuts

Add a `useEffect` that registers a `keydown` listener on the document:

```typescript
useEffect(() => {
  if (userRole === "billing_manager") return; // read-only roles skip
  const onKey = (e: KeyboardEvent) => {
    if (e.target instanceof HTMLInputElement || e.target instanceof HTMLTextAreaElement) return;
    if (e.metaKey || e.ctrlKey || e.altKey) return;
    if (e.key === "a" || e.key === "A") {
      if (allGreen) handleApprove();
    } else if (e.key === "r" || e.key === "R") {
      if (focusedFindingId != null) openRouteDialog(focusedFindingId);
    } else if (e.key === "j" || e.key === "J") {
      focusNextFinding();
    } else if (e.key === "k" || e.key === "K") {
      focusPrevFinding();
    }
  };
  window.addEventListener("keydown", onKey);
  return () => window.removeEventListener("keydown", onKey);
}, [allGreen, focusedFindingId, userRole]);
```

`focusNextFinding` / `focusPrevFinding` advance through the page's
finding cards using a focused-id state. `openRouteDialog` triggers the
existing route dialog the wizard surface provides. If the wizard owns
those handlers (it does, per
`components/manager/wet-check-wizard.tsx`), expose them via a callback
prop or via a ref, do NOT reimplement.

#### (5) Optimistic approve mutation

Find the existing approve mutation in `wet-check-review.tsx` (it calls
`POST /api/wet-checks/:id/approve`). Convert to use react-query's
`onMutate` for an optimistic state flip:

```typescript
const approveMutation = useMutation({
  mutationFn: async () => apiRequest(`/api/wet-checks/${id}/approve`, "POST", {}),
  onMutate: async () => {
    // Cancel inflight queries that could overwrite optimistic state.
    await queryClient.cancelQueries({ queryKey: ["/api/wet-checks", id] });
    const previous = queryClient.getQueryData(["/api/wet-checks", id]);
    queryClient.setQueryData(["/api/wet-checks", id], (old: any) => ({
      ...old,
      status: "approved",
      approvedAt: new Date().toISOString(),
    }));
    return { previous };
  },
  onError: (_err, _vars, ctx) => {
    if (ctx?.previous) queryClient.setQueryData(["/api/wet-checks", id], ctx.previous);
    toast({ title: "Approve failed", description: "Reverted. Please try again.", variant: "destructive" });
  },
  onSettled: () => {
    queryClient.invalidateQueries({ queryKey: ["/api/wet-checks", id] });
    queryClient.invalidateQueries({ queryKey: ["/api/wet-check-billings"] });
  },
});
```

The UI flips to "approved" instantly; if the server rejects, state
rolls back and a toast surfaces the failure.

### Feels-fast lock — `wet-check-review-perf.test.tsx`

```typescript
describe("wet-check-review page — feels-fast contract (Slice 6)", () => {
  it("first interactive paint < 800ms on a primed cache", async () => {
    primeCache({ wetCheckId: 42, allGreen: true });
    const t0 = performance.now();
    render(<WetCheckReviewPage params={{ id: "42" }} />);
    await screen.findByText("Approve & pass to billing");
    const t1 = performance.now();
    expect(t1 - t0).toBeLessThan(800);
  });

  it("approve mutation flips UI before server resolves (optimistic)", async () => {
    const slowFetch = mockFetchWithDelay(200);
    primeCache({ wetCheckId: 42, allGreen: true });
    render(<WetCheckReviewPage params={{ id: "42" }} />);
    fireEvent.keyDown(window, { key: "A" });
    // Within 50ms of keypress, the status badge should already read "approved"
    await waitFor(() => {
      expect(screen.getByTestId("status-badge")).toHaveTextContent(/approved/i);
    }, { timeout: 50 });
  });

  it("keyboard-only review flow takes ≤ 3 keystrokes from an all-green load", async () => {
    primeCache({ wetCheckId: 42, allGreen: true });
    render(<WetCheckReviewPage params={{ id: "42" }} />);
    await screen.findByText("Approve & pass to billing");
    let keystrokes = 0;
    fireEvent.keyDown(window, { key: "A" }); keystrokes++;
    await waitFor(() => expect(screen.getByTestId("status-badge")).toHaveTextContent(/approved/i));
    expect(keystrokes).toBeLessThanOrEqual(3);
  });
});
```

The 800ms target assumes a primed query cache (i.e., the user has
visited /wet-checks/pending-review immediately before). If the test
runs slower than 800ms on CI hardware, raise the bound — but the
relative target stays "fast enough that there's no skeleton flash."

### Acceptance criteria

- `DismissibleHelp` exists at `components/shared/dismissible-help.tsx`
  with `isHelpDismissed` / `resetHelpDismissal` helpers exported.
- The primitive tests cover open/dismiss/Esc/persistence.
- The review page renders a 3-step workflow indicator that highlights
  the current step.
- An all-green DismissibleHelp banner appears when zones and findings
  are all resolved, mentioning the `A` shortcut.
- Keyboard shortcuts `A` / `R` / `J` / `K` work as specified, with
  input fields and modifier keys excluded.
- Approve mutation is optimistic — UI flips to "approved" before the
  server confirms; rollback on failure with a toast.
- `billing_manager` role sees a "View only" banner and cannot trigger
  approve via keyboard or button.
- "Show help" toggle resets dismissed guides on demand.
- All perf-test bounds in `wet-check-review-perf.test.tsx` pass on
  CI hardware. (Adjust the 800ms bound if needed for CI variance, but
  document the adjustment in the PR.)
- All existing review page tests pass.
