/* -------------------------------------------------------------------------- */
/* Antigravity-style ring particle field                                      */
/*                                                                            */
/* Mimics Bramus's css-houdini-ringparticles worklet (the one Google's        */
/* antigravity.google site uses):                                             */
/*   - a BAND of small dots (multiple rows across the band thickness)         */
/*   - the band rotates on its own (Z-axis spin)                              */
/*   - the band tilts on X and Y axes (gyroscope wobble in 3D)                */
/*   - the band's edge morphs via overlapping sine waves                      */
/*   - cursor velocity stretches the band like a comet                        */
/*   - smooth cursor follow with inertia                                      */
/*                                                                            */
/* Rendered with a pre-rendered radial-gradient sprite + drawImage so we can  */
/* afford ~600 particles at 60fps.                                            */
/* -------------------------------------------------------------------------- */

function RingParticleField() {
  const canvasRef = useRef<HTMLCanvasElement>(null);
  const reduced =
    typeof window !== "undefined" &&
    window.matchMedia("(prefers-reduced-motion: reduce)").matches;
  const isMobile =
    typeof window !== "undefined" &&
    window.matchMedia("(max-width: 640px)").matches;

  const cfg = useMemo(() => {
    const baseRadius = isMobile ? 95 : 135;
    const thickness = isMobile ? 32 : 46;
    const rows = isMobile ? 5 : 7;
    const perRow = isMobile ? 60 : 92;
    return { baseRadius, thickness, rows, perRow };
  }, [isMobile]);

  // Particles in band-local polar coords (z=0 in local space; depth comes from tilt).
  const particles = useMemo(() => {
    const out: Array<{ a: number; r: number; size: number; alpha: number }> = [];
    for (let row = 0; row < cfg.rows; row++) {
      const t = cfg.rows === 1 ? 0 : row / (cfg.rows - 1); // 0..1
      const radialBase = cfg.baseRadius + (t - 0.5) * cfg.thickness;
      const angleOffset = (row * Math.PI) / cfg.perRow; // stagger rows so dots interleave
      for (let i = 0; i < cfg.perRow; i++) {
        const a = (i / cfg.perRow) * Math.PI * 2 + angleOffset;
        const jitter = ((i * 53 + row * 17) % 11) - 5;
        const r = radialBase + jitter * 0.45;
        const edgeFalloff = 1 - Math.abs(t - 0.5) * 1.4; // brighter mid-band, soft edges
        const alpha = Math.max(0.2, 0.7 * edgeFalloff);
        const size = 0.9 + Math.random() * 1.2; // 0.9..2.1 px (pre-depth)
        out.push({ a, r, size, alpha });
      }
    }
    return out;
    // eslint-disable-next-line react-hooks/exhaustive-deps
  }, [cfg]);

  useEffect(() => {
    if (reduced) return;
    const canvas = canvasRef.current;
    if (!canvas) return;
    const ctx = canvas.getContext("2d", { alpha: true });
    if (!ctx) return;

    const dpr = Math.max(1, Math.min(2, window.devicePixelRatio || 1));
    let w = window.innerWidth;
    let h = window.innerHeight;

    const resize = () => {
      w = window.innerWidth;
      h = window.innerHeight;
      canvas.width = Math.floor(w * dpr);
      canvas.height = Math.floor(h * dpr);
      canvas.style.width = `${w}px`;
      canvas.style.height = `${h}px`;
      ctx.setTransform(dpr, 0, 0, dpr, 0, 0);
    };
    resize();
    window.addEventListener("resize", resize);

    // Pre-render a particle stamp once. drawImage of this is ~10× faster than
    // re-creating a radial gradient per particle per frame.
    const STAMP = 64;
    const stamp = document.createElement("canvas");
    stamp.width = stamp.height = STAMP;
    const sctx = stamp.getContext("2d")!;
    const sg = sctx.createRadialGradient(
      STAMP / 2,
      STAMP / 2,
      0,
      STAMP / 2,
      STAMP / 2,
      STAMP / 2
    );
    sg.addColorStop(0, "rgba(30,64,175,1)"); // blue-800 core
    sg.addColorStop(0.18, "rgba(37,99,235,0.85)"); // blue-600
    sg.addColorStop(0.45, "rgba(96,165,250,0.35)"); // soft halo
    sg.addColorStop(1, "rgba(96,165,250,0)");
    sctx.fillStyle = sg;
    sctx.fillRect(0, 0, STAMP, STAMP);

    // Cursor + smoothed band-center + velocity
    const target = { x: w / 2, y: h / 2, active: false };
    const cur = { x: w / 2, y: h / 2 };
    const lastCur = { x: cur.x, y: cur.y };
    const vel = { x: 0, y: 0 };
    let visible = 0;
    let rotZ = 0;

    const onMove = (e: PointerEvent) => {
      target.x = e.clientX;
      target.y = e.clientY;
      if (!target.active) {
        cur.x = e.clientX;
        cur.y = e.clientY;
        lastCur.x = cur.x;
        lastCur.y = cur.y;
      }
      target.active = true;
    };
    const onLeave = () => {
      target.active = false;
    };
    window.addEventListener("pointermove", onMove);
    window.addEventListener("pointerleave", onLeave);

    let raf = 0;
    let last = performance.now();
    const PERSPECTIVE = 720;

    const tick = (now: number) => {
      const dt = Math.min(64, now - last);
      last = now;

      // Smooth cursor follow
      const k = 0.16;
      cur.x += (target.x - cur.x) * k;
      cur.y += (target.y - cur.y) * k;

      // Low-pass-filtered velocity for stretch warp
      vel.x = vel.x * 0.82 + (cur.x - lastCur.x) * 0.18;
      vel.y = vel.y * 0.82 + (cur.y - lastCur.y) * 0.18;
      lastCur.x = cur.x;
      lastCur.y = cur.y;

      // Visibility fade in/out
      const visTarget = target.active ? 1 : 0;
      visible += (visTarget - visible) * 0.08;

      // Z-spin + gyroscope tilt on X & Y (multi-freq sines = non-repeating wobble)
      rotZ += 0.00028 * dt;
      const tiltX =
        Math.sin(now * 0.00028) * 0.55 + Math.sin(now * 0.00047 + 1.7) * 0.22;
      const tiltY =
        Math.cos(now * 0.00033) * 0.55 + Math.cos(now * 0.00061 + 0.9) * 0.22;

      const vmag = Math.hypot(vel.x, vel.y);
      const vAngle = Math.atan2(vel.y, vel.x);
      const stretch = Math.min(0.42, vmag * 0.014);

      ctx.clearRect(0, 0, w, h);

      if (visible <= 0.01) {
        raf = requestAnimationFrame(tick);
        return;
      }

      ctx.save();
      ctx.globalAlpha = visible;
      ctx.translate(cur.x, cur.y);

      // Soft cursor halo (flat behind the band)
      const halo = ctx.createRadialGradient(
        0,
        0,
        0,
        0,
        0,
        cfg.baseRadius + cfg.thickness * 1.4
      );
      halo.addColorStop(0, "rgba(96,165,250,0.12)");
      halo.addColorStop(0.55, "rgba(125,211,252,0.06)");
      halo.addColorStop(1, "rgba(125,211,252,0)");
      ctx.fillStyle = halo;
      ctx.beginPath();
      ctx.arc(0, 0, cfg.baseRadius + cfg.thickness * 1.4, 0, Math.PI * 2);
      ctx.fill();

      const cosTX = Math.cos(tiltX);
      const sinTX = Math.sin(tiltX);
      const cosTY = Math.cos(tiltY);
      const sinTY = Math.sin(tiltY);

      // Additive so overlapping glows accumulate (gives a denser inner band)
      ctx.globalCompositeOperation = "lighter";

      for (let i = 0; i < particles.length; i++) {
        const p = particles[i];
        const a = p.a + rotZ;

        // Shape morph — overlapping sines on the radius
        const wave =
          Math.sin(p.a * 3 + now * 0.0009) * 5.5 +
          Math.sin(p.a * 5 - now * 0.0013 + 1.1) * 3.5 +
          Math.sin(p.a * 2 + now * 0.0006) * 2.2;

        // Velocity warp — particles aligned with motion get pulled out, opposite squeezed in
        const align = Math.cos(a - vAngle); // -1..1
        const warpR = stretch * cfg.baseRadius * align;

        const r = p.r + wave + warpR;

        // 3D position in band-local space (z=0 before tilt)
        const x0 = Math.cos(a) * r;
        const y0 = Math.sin(a) * r;
        // Rotate around X axis: y' and z' (z0 = 0)
        const y1 = y0 * cosTX;
        const z1 = y0 * sinTX;
        // Rotate around Y axis: x' and z'
        const x2 = x0 * cosTY + z1 * sinTY;
        const z2 = -x0 * sinTY + z1 * cosTY;
        const y2 = y1;

        // Perspective projection
        const scale = PERSPECTIVE / (PERSPECTIVE - z2);
        const px = x2 * scale;
        const py = y2 * scale;

        // Depth shading: front bigger/brighter, back smaller/dimmer
        const depth = Math.max(0.35, Math.min(1.4, scale));
        const finalAlpha = Math.min(1, p.alpha * depth);
        const finalSize = p.size * depth;

        const drawSize = finalSize * 7; // includes glow halo around the core
        ctx.globalAlpha = visible * finalAlpha;
        ctx.drawImage(stamp, px - drawSize / 2, py - drawSize / 2, drawSize, drawSize);
      }

      ctx.restore();
      raf = requestAnimationFrame(tick);
    };
    raf = requestAnimationFrame(tick);

    return () => {
      cancelAnimationFrame(raf);
      window.removeEventListener("resize", resize);
      window.removeEventListener("pointermove", onMove);
      window.removeEventListener("pointerleave", onLeave);
    };
  }, [cfg, particles, reduced]);

  return (
    <canvas
      ref={canvasRef}
      aria-hidden
      className="pointer-events-none fixed inset-0 z-[1]"
    />
  );
}

export default function Login() {
  const [credentials, setCredentials] = useState<LoginCredentials>({
    username: "",
    password: "",
  });
  const [isLoading, setIsLoading] = useState(false);
  const [emailVerificationNeeded, setEmailVerificationNeeded] = useState<string | null>(null);
  const [sendingVerification, setSendingVerification] = useState(false);
  const { toast } = useToast();

  const handleLogin = async (e: React.FormEvent) => {
    e.preventDefault();
    setIsLoading(true);
    try {
      const user = await apiRequest("/api/auth/login", "POST", credentials);
      safeSet("user", JSON.stringify(user));
      toast({
        title: "Login Successful",
        description: `Welcome back, ${user.name}!`,
      });
      window.location.href = "/";
    } catch (error: any) {
      if (
        (error.message && error.message.includes("email verification")) ||
        error.message?.includes("verify your email")
      ) {
        setEmailVerificationNeeded(credentials.username);
        toast({
          title: "Email Verification Required",
          description: "Please verify your email address before logging in.",
          variant: "destructive",
        });
      } else {
        toast({
          title: "Login Failed",
          description: error.message || "Invalid credentials",
          variant: "destructive",
        });
      }
    } finally {
      setIsLoading(false);
    }
  };

  const handleResendVerification = async () => {
    if (!emailVerificationNeeded) return;
    setSendingVerification(true);
    try {
      await apiRequest("/api/auth/resend-verification", "POST", {
        email: emailVerificationNeeded,
      });
      toast({
        title: "Verification Email Sent",
        description: "Please check your email for the verification link.",
      });
    } catch (error: any) {
      toast({
        title: "Error",
        description: error.message || "Failed to send verification email",
        variant: "destructive",
      });
    } finally {
      setSendingVerification(false);
    }
  };

  return (
    <div className="relative min-h-screen w-full overflow-hidden bg-[#f5fbff] text-slate-900 flex flex-col">
      <StaticBackdrop />
      <RingParticleField />

      <div className="pointer-events-auto relative z-10 flex-1 flex items-center justify-center p-5 md:p-8">
        <Card className="w-full max-w-md bg-white/85 backdrop-blur-xl border border-sky-100 shadow-[0_20px_60px_-15px_rgba(59,130,246,0.25)] text-slate-900 rounded-3xl">
          <CardHeader className="text-center pt-8 pb-4">
            <div className="flex justify-center mb-6">
              <img
                src={irrigoProLogo}
                alt="IrrigoPro Logo"
                className="h-28 w-auto drop-shadow-2xl"
              />
            </div>
            <CardTitle className="text-3xl font-bold text-slate-900 tracking-tight">
              IrrigoPro
            </CardTitle>
            <p className="text-blue-700 mt-2 text-base">
              Professional irrigation management
            </p>
          </CardHeader>
          <CardContent className="px-6 pb-8">
            <form onSubmit={handleLogin} className="space-y-5">
              <div className="space-y-2">
                <Label
                  htmlFor="username"
                  className="text-slate-700 text-sm font-medium"
                >
                  Username
                </Label>
                <Input
                  id="username"
                  type="text"
                  placeholder="Enter your username"
                  value={credentials.username}
                  onChange={(e) =>
                    setCredentials({ ...credentials, username: e.target.value })
                  }
                  required
                  data-testid="input-username"
                  className="h-14 bg-white/90 border-2 border-slate-200 text-slate-900 placeholder-slate-400 rounded-xl text-base focus:border-sky-500 focus:ring-2 focus:ring-sky-400/30"
                />
              </div>

              <div className="space-y-2">
                <Label
                  htmlFor="password"
                  className="text-slate-700 text-sm font-medium"
                >
                  Password
                </Label>
                <Input
                  id="password"
                  type="password"
                  placeholder="Password"
                  value={credentials.password}
                  onChange={(e) =>
                    setCredentials({ ...credentials, password: e.target.value })
                  }
                  required
                  data-testid="input-password"
                  className="h-14 bg-white/90 border-2 border-slate-200 text-slate-900 placeholder-slate-400 rounded-xl text-base focus:border-sky-500 focus:ring-2 focus:ring-sky-400/30"
                />
              </div>

              <Button
                type="submit"
                size="lg"
                data-testid="button-login"
                className="w-full h-14 bg-gradient-to-r from-sky-500 to-blue-600 hover:from-sky-600 hover:to-blue-700 text-white font-semibold text-lg rounded-xl shadow-lg shadow-sky-500/30 transition-all duration-200 mt-2"
                disabled={isLoading}
              >
                {isLoading ? (
                  <Lock className="w-5 h-5 mr-2 animate-spin" />
                ) : (
                  <LogIn className="w-5 h-5 mr-2" />
                )}
                Sign In
              </Button>
            </form>

            {emailVerificationNeeded && (
              <div className="mt-5 p-5 bg-amber-50 border border-amber-300 rounded-2xl">
                <div className="flex items-center mb-3">
                  <Mail className="w-5 h-5 mr-2 text-amber-600" />
                  <span className="text-base text-amber-800 font-semibold">
                    Email Verification Required
                  </span>
                </div>
                <p className="text-sm text-amber-700 mb-4">
                  Please verify your email address to access your account.
                </p>
                <Button
                  onClick={handleResendVerification}
                  disabled={sendingVerification}
                  data-testid="button-resend-verification"
                  className="w-full h-12 bg-amber-500 hover:bg-amber-600 text-white rounded-xl"
                >
                  {sendingVerification ? (
                    <Lock className="w-4 h-4 mr-2 animate-spin" />
                  ) : (
                    <Mail className="w-4 h-4 mr-2" />
                  )}
                  Resend Verification Email
                </Button>
              </div>
            )}

            <div className="mt-6 text-center">
              <Button
                variant="link"
                onClick={() => (window.location.href = "/forgot-password")}
                data-testid="link-forgot-password"
                className="text-base text-blue-700 hover:text-blue-900 transition-colors"
              >
                Forgot your password?
              </Button>
            </div>
          </CardContent>
        </Card>
      </div>

      <div className="relative z-10">
        <PoweredByFooter />
      </div>
    </div>
  );
}

Knobs to taste (all inside RingParticleField):

baseRadius / thickness / rows / perRow in cfg — band geometry and density
0.55 and 0.22 in tiltX/tiltY — gyroscope tilt amplitude (radians); raise to ~0.8 for more dramatic flipping
0.00028 for rotZ — z-axis spin speed
PERSPECTIVE = 720 — lower = stronger 3D foreshortening
Wave amplitudes 5.5, 3.5, 2.2 — edge morph intensity
0.014 in stretch — cursor-drag elongation strength
k = 0.16 — cursor follow snappiness; lower = more inertia/lag
