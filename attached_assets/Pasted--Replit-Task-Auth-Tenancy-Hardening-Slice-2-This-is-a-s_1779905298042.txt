**Concrete changes.**

### 1. Add `requireSameCompanyAsBillingSheet` middleware

Place immediately after `requireSameCompanyAsWorkOrder` (introduced in
Slice 1). Mirror its structure exactly:

```ts
// Tenant guard for billing-sheet routes. Same contract as
// requireSameCompanyAsWorkOrder: load by req.params.id, derive
// companyId via the customer link, 404 on mismatch, super-admin
// bypass, audit-log cross-tenant attempts.
//
// Until Slice 4 lands the billing_sheets.companyId column, the row's
// tenant is read through customerId → customers.companyId.
const requireSameCompanyAsBillingSheet = async (req: any, res: any, next: any) => {
  const userRole = req.authenticatedUserRole;
  if (userRole === 'super_admin') {
    return next();
  }
  const userCompanyId = req.authenticatedUserCompanyId;
  if (userCompanyId == null) {
    res.status(403).json({ message: "Company scope required for this action." });
    return;
  }
  const billingSheetId = parseInt(req.params.id);
  if (!Number.isFinite(billingSheetId) || billingSheetId <= 0) {
    res.status(400).json({ message: "Invalid billing sheet ID" });
    return;
  }
  try {
    const billingSheet = await storage.getBillingSheetById(billingSheetId);
    if (!billingSheet) {
      res.status(404).json({ message: "Billing sheet not found" });
      return;
    }
    const customer = billingSheet.customerId
      ? await storage.getCustomer(billingSheet.customerId)
      : null;
    const billingSheetCompanyId = customer?.companyId ?? null;
    if (billingSheetCompanyId == null || billingSheetCompanyId !== userCompanyId) {
      res.status(404).json({ message: "Billing sheet not found" });
      console.warn(
        `[AUDIT] cross_tenant_billing_sheet_access ` +
        `actor=${req.authenticatedUserId} actorCompany=${userCompanyId} ` +
        `targetBillingSheet=${billingSheetId} targetCompany=${billingSheetCompanyId ?? 'null'} ` +
        `route=${req.method} ${req.path}`
      );
      return;
    }
    req.tenantScopedBillingSheet = billingSheet;
    next();
  } catch (error) {
    console.error('Error checking billing-sheet tenant scope:', error);
    res.status(500).json({ message: "Couldn't verify tenant scope — please retry." });
    return;
  }
};
```

### 3. Verify `GET /api/work-orders/:id/billing-sheets` doesn't bypass

This route was wrapped with `requireSameCompanyAsWorkOrder` in
Slice 1 — the parent work order is the tenant anchor. Confirm the
handler at routes.ts:12988 reads sheets only for the scoped work order

not enumerate sheets across tenants. If it does, the handler is
already correct because the parent guard rejected before it ran. No
change required in Slice 2; flag if otherwise.
