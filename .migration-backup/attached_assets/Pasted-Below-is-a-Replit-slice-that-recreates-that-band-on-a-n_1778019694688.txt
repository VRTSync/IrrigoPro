/**
 * Antigravity-style ring particles, recreated in Canvas.
 * Mimics Bramus's css-houdini-ringparticles worklet that powers antigravity.google:
 *   - a BAND (ring with thickness) of many tiny dots
 *   - multiple particle rows distributed across the band's thickness
 *   - the whole band slowly rotates and follows the cursor with inertia
 */
function RingParticleField() {
  const canvasRef = useRef<HTMLCanvasElement>(null);
  const reduced =
    typeof window !== "undefined" &&
    window.matchMedia("(prefers-reduced-motion: reduce)").matches;
  const isMobile =
    typeof window !== "undefined" &&
    window.matchMedia("(max-width: 640px)").matches;

  // Tuneable (mirrors --ring-radius, --ring-thickness, --particle-rows, --particle-count from the worklet)
  const cfg = useMemo(() => {
    const baseRadius = isMobile ? 90 : 130;
    const thickness = isMobile ? 28 : 40;
    const rows = isMobile ? 4 : 6;
    const perRow = isMobile ? 56 : 88;
    return { baseRadius, thickness, rows, perRow };
  }, [isMobile]);

  // Pre-build particles once (positions in band-local polar coords).
  // Each particle gets a fixed angle + radius offset so the band looks dense and organic
  // rather than gridded.
  const particles = useMemo(() => {
    const out: Array<{ a: number; r: number; size: number; alpha: number }> = [];
    for (let row = 0; row < cfg.rows; row++) {
      // Distribute rows across the band thickness (-thickness/2 .. +thickness/2)
      const t = cfg.rows === 1 ? 0 : row / (cfg.rows - 1); // 0..1
      const radialBase = cfg.baseRadius + (t - 0.5) * cfg.thickness;
      // Stagger row angle so dots interleave instead of sitting on radial spokes
      const angleOffset = (row * Math.PI) / cfg.perRow;
      for (let i = 0; i < cfg.perRow; i++) {
        const a = (i / cfg.perRow) * Math.PI * 2 + angleOffset;
        // tiny per-particle radial jitter for organic feel
        const jitter = ((i * 53 + row * 17) % 11) - 5;
        const r = radialBase + jitter * 0.45;
        // Edge rows fade slightly so the band has soft edges
        const edgeFalloff = 1 - Math.abs(t - 0.5) * 1.4; // brighter in middle
        const alpha = Math.max(0.18, 0.65 * edgeFalloff);
        const size = 0.9 + Math.random() * 1.1; // 0.9 .. 2.0 px
        out.push({ a, r, size, alpha });
      }
    }
    return out;
    // eslint-disable-next-line react-hooks/exhaustive-deps
  }, [cfg]);

  useEffect(() => {
    if (reduced) return;
    const canvas = canvasRef.current;
    if (!canvas) return;
    const ctx = canvas.getContext("2d", { alpha: true });
    if (!ctx) return;

    let raf = 0;
    const dpr = Math.max(1, Math.min(2, window.devicePixelRatio || 1));
    let w = window.innerWidth;
    let h = window.innerHeight;

    const resize = () => {
      w = window.innerWidth;
      h = window.innerHeight;
      canvas.width = Math.floor(w * dpr);
      canvas.height = Math.floor(h * dpr);
      canvas.style.width = `${w}px`;
      canvas.style.height = `${h}px`;
      ctx.setTransform(dpr, 0, 0, dpr, 0, 0);
    };
    resize();
    window.addEventListener("resize", resize);

    // Cursor + smoothed band-center
    const target = { x: w / 2, y: h / 2, active: false };
    const cur = { x: w / 2, y: h / 2 };
    let visible = 0; // 0..1 fade
    let rotation = 0; // band rotation (radians)

    const onMove = (e: PointerEvent) => {
      target.x = e.clientX;
      target.y = e.clientY;
      if (!target.active) {
        cur.x = e.clientX;
        cur.y = e.clientY;
      }
      target.active = true;
    };
    const onLeave = () => {
      target.active = false;
    };
    window.addEventListener("pointermove", onMove);
    window.addEventListener("pointerleave", onLeave);

    let last = performance.now();
    const tick = (now: number) => {
      const dt = Math.min(64, now - last);
      last = now;

      // Smooth cursor follow (gives the "drag the field with you" feel)
      const k = 0.16;
      cur.x += (target.x - cur.x) * k;
      cur.y += (target.y - cur.y) * k;

      // Visibility fade in/out
      const visTarget = target.active ? 1 : 0;
      visible += (visTarget - visible) * 0.08;

      // Band rotation
      rotation += 0.00018 * dt; // slow

      ctx.clearRect(0, 0, w, h);

      if (visible > 0.01) {
        ctx.save();
        ctx.globalAlpha = visible;
        ctx.translate(cur.x, cur.y);
        ctx.rotate(rotation);

        // Soft cursor halo behind the band
        const halo = ctx.createRadialGradient(0, 0, 0, 0, 0, cfg.baseRadius + cfg.thickness);
        halo.addColorStop(0, "rgba(96,165,250,0.10)");
        halo.addColorStop(0.55, "rgba(125,211,252,0.06)");
        halo.addColorStop(1, "rgba(125,211,252,0)");
        ctx.fillStyle = halo;
        ctx.beginPath();
        ctx.arc(0, 0, cfg.baseRadius + cfg.thickness, 0, Math.PI * 2);
        ctx.fill();

        // Particles — additive-ish soft dots
        // Use 'lighter' for a subtle glow accumulation between adjacent dots
        ctx.globalCompositeOperation = "source-over";
        for (let i = 0; i < particles.length; i++) {
          const p = particles[i];
          const x = Math.cos(p.a) * p.r;
          const y = Math.sin(p.a) * p.r;

          // Soft outer glow
          const g = ctx.createRadialGradient(x, y, 0, x, y, p.size * 3.2);
          g.addColorStop(0, `rgba(59,130,246,${p.alpha * 0.55})`);
          g.addColorStop(0.5, `rgba(96,165,250,${p.alpha * 0.22})`);
          g.addColorStop(1, "rgba(96,165,250,0)");
          ctx.fillStyle = g;
          ctx.beginPath();
          ctx.arc(x, y, p.size * 3.2, 0, Math.PI * 2);
          ctx.fill();

          // Crisp core
          ctx.fillStyle = `rgba(30,64,175,${p.alpha})`; // blue-800
          ctx.beginPath();
          ctx.arc(x, y, p.size, 0, Math.PI * 2);
          ctx.fill();
        }

        ctx.restore();
      }

      raf = requestAnimationFrame(tick);
    };
    raf = requestAnimationFrame(tick);

    return () => {
      cancelAnimationFrame(raf);
      window.removeEventListener("resize", resize);
      window.removeEventListener("pointermove", onMove);
      window.removeEventListener("pointerleave", onLeave);
    };
  }, [cfg, particles, reduced]);

  return (
    <canvas
      ref={canvasRef}
      aria-hidden
      className="pointer-events-none fixed inset-0 z-[1]"
    />
  );
}

/** Static near-white-with-blue-tint background (no per-frame work). */
function StaticBackdrop() {
  const style: CSSProperties = {
    background:
      "radial-gradient(140% 100% at 50% 0%, #ffffff 0%, #f5fbff 38%, #eaf6ff 72%, #dceffd 100%)",
  };
  return (
    <div aria-hidden className="absolute inset-0 z-0 overflow-hidden">
      <div className="absolute inset-0" style={style} />
      {/* very faint sky aurora — pure CSS, no animation */}
      <div
        className="absolute -top-32 -left-20 h-[520px] w-[520px] rounded-full blur-3xl opacity-60"
        style={{ background: "radial-gradient(circle at 30% 30%, rgba(186,230,253,0.55), transparent 65%)" }}
      />
      <div
        className="absolute top-1/3 -right-24 h-[460px] w-[460px] rounded-full blur-3xl opacity-50"
        style={{ background: "radial-gradient(circle at 30% 30%, rgba(147,197,253,0.45), transparent 65%)" }}
      />
      <div
        className="absolute -bottom-24 left-1/4 h-[560px] w-[560px] rounded-full blur-3xl opacity-50"
        style={{ background: "radial-gradient(circle at 30% 30%, rgba(165,243,252,0.45), transparent 65%)" }}
      />
      {/* soft edge vignette */}
      <div className="absolute inset-0 bg-[radial-gradient(80%_60%_at_50%_50%,rgba(255,255,255,0)_55%,rgba(186,230,253,0.18)_100%)]" />
    </div>
  );
}

export default function Login() {
  const [credentials, setCredentials] = useState<LoginCredentials>({ username: "", password: "" });
  const [isLoading, setIsLoading] = useState(false);
  const [emailVerificationNeeded, setEmailVerificationNeeded] = useState<string | null>(null);
  const [sendingVerification, setSendingVerification] = useState(false);
  const { toast } = useToast();

  const handleLogin = async (e: React.FormEvent) => {
    e.preventDefault();
    setIsLoading(true);
    try {
      const user = await apiRequest("/api/auth/login", "POST", credentials);
      safeSet("user", JSON.stringify(user));
      toast({ title: "Login Successful", description: `Welcome back, ${user.name}!` });
      window.location.href = "/";
    } catch (error: any) {
      if ((error.message && error.message.includes("email verification")) || error.message?.includes("verify your email")) {
        setEmailVerificationNeeded(credentials.username);
        toast({
          title: "Email Verification Required",
          description: "Please verify your email address before logging in.",
          variant: "destructive",
        });
      } else {
        toast({
          title: "Login Failed",
          description: error.message || "Invalid credentials",
          variant: "destructive",
        });
      }
    } finally {
      setIsLoading(false);
    }
  };

  const handleResendVerification = async () => {
    if (!emailVerificationNeeded) return;
    setSendingVerification(true);
    try {
      await apiRequest("/api/auth/resend-verification", "POST", { email: emailVerificationNeeded });
      toast({ title: "Verification Email Sent", description: "Please check your email for the verification link." });
    } catch (error: any) {
      toast({
        title: "Error",
        description: error.message || "Failed to send verification email",
        variant: "destructive",
      });
    } finally {
      setSendingVerification(false);
    }
  };

  return (
    <div className="relative min-h-screen w-full overflow-hidden bg-[#f5fbff] text-slate-900 flex flex-col">
      <StaticBackdrop />
      <RingParticleField />

      <div className="pointer-events-auto relative z-10 flex-1 flex items-center justify-center p-5 md:p-8">
        <Card className="w-full max-w-md bg-white/85 backdrop-blur-xl border border-sky-100 shadow-[0_20px_60px_-15px_rgba(59,130,246,0.25)] text-slate-900 rounded-3xl">
          <CardHeader className="text-center pt-8 pb-4">
            <div className="flex justify-center mb-6">
              <img src={irrigoProLogo} alt="IrrigoPro Logo" className="h-28 w-auto drop-shadow-2xl" />
            </div>
            <CardTitle className="text-3xl font-bold text-slate-900 tracking-tight">IrrigoPro</CardTitle>
            <p className="text-blue-700 mt-2 text-base">Professional irrigation management</p>
          </CardHeader>
          <CardContent className="px-6 pb-8">
            <form onSubmit={handleLogin} className="space-y-5">
              <div className="space-y-2">
                <Label htmlFor="username" className="text-slate-700 text-sm font-medium">Username</Label>
                <Input
                  id="username"
                  type="text"
                  placeholder="Enter your username"
                  value={credentials.username}
                  onChange={(e) => setCredentials({ ...credentials, username: e.target.value })}
                  required
                  data-testid="input-username"
                  className="h-14 bg-white/90 border-2 border-slate-200 text-slate-900 placeholder-slate-400 rounded-xl text-base focus:border-sky-500 focus:ring-2 focus:ring-sky-400/30"
                />
              </div>

              <div className="space-y-2">
                <Label htmlFor="password" className="text-slate-700 text-sm font-medium">Password</Label>
                <Input
                  id="password"
                  type="password"
                  placeholder="Password"
                  value={credentials.password}
                  onChange={(e) => setCredentials({ ...credentials, password: e.target.value })}
                  required
                  data-testid="input-password"
                  className="h-14 bg-white/90 border-2 border-slate-200 text-slate-900 placeholder-slate-400 rounded-xl text-base focus:border-sky-500 focus:ring-2 focus:ring-sky-400/30"
                />
              </div>

              <Button
                type="submit"
                size="lg"
                data-testid="button-login"
                className="w-full h-14 bg-gradient-to-r from-sky-500 to-blue-600 hover:from-sky-600 hover:to-blue-700 text-white font-semibold text-lg rounded-xl shadow-lg shadow-sky-500/30 transition-all duration-200 mt-2"
                disabled={isLoading}
              >
                {isLoading ? <Lock className="w-5 h-5 mr-2 animate-spin" /> : <LogIn className="w-5 h-5 mr-2" />}
                Sign In
              </Button>
            </form>

            {emailVerificationNeeded && (
              <div className="mt-5 p-5 bg-amber-50 border border-amber-300 rounded-2xl">
                <div className="flex items-center mb-3">
                  <Mail className="w-5 h-5 mr-2 text-amber-600" />
                  <span className="text-base text-amber-800 font-semibold">Email Verification Required</span>
                </div>
                <p className="text-sm text-amber-700 mb-4">
                  Please verify your email address to access your account.
                </p>
                <Button
                  onClick={handleResendVerification}
                  disabled={sendingVerification}
                  data-testid="button-resend-verification"
                  className="w-full h-12 bg-amber-500 hover:bg-amber-600 text-white rounded-xl"
                >
                  {sendingVerification ? <Lock className="w-4 h-4 mr-2 animate-spin" /> : <Mail className="w-4 h-4 mr-2" />}
                  Resend Verification Email
                </Button>
              </div>
            )}

            <div className="mt-6 text-center">
              <Button
                variant="link"
                onClick={() => (window.location.href = "/forgot-password")}
                data-testid="link-forgot-password"
                className="text-base text-blue-700 hover:text-blue-900 transition-colors"
              >
                Forgot your password?
              </Button>
            </div>
          </CardContent>
        </Card>
      </div>

      <div className="relative z-10">
        <PoweredByFooter />
      </div>
    </div>
  );
}

Knobs to tweak in Replit if you want it tighter/looser:

baseRadius — overall band size
thickness + rows — fatter band / more depth
perRow — density (raise for chunkier band)
k (line ~136) — follow snappiness; lower = more "drag/inertia" feel
0.00018 — rotation speed
